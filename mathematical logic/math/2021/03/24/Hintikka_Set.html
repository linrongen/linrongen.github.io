定义 1.10

\[H_A(v(P_1), v(P_2), \dots, v(P_n)) = \hat{v}(A),\]

其中 $FV(A)=\{P_1, P_2, \dots, P_n\}$.



定理 1.12

令 $v$ 满足 $v(P_i) = x_i$ 对 $1\le i\le n$ 成立. 再令

\[A=\bigvee\limits_{i=1}^{m} \bigwedge\limits_{k=1}^{n} P_{i,k}^{\star}, \qquad A'=\bigwedge\limits_{j=1}^{m} \bigvee\limits_{k=1}^{n} Q_{j,k}^{\star},\]

其中

\[P_{i,k}^{\star} = \begin{cases} P_k & \text{if} \;\; a_{ik} = \mathrm{T} \\ \lnot P_k & \text{if} \;\; a_{ik} = \mathrm{F} \end{cases}, \qquad Q_{j,k}^{\star} = \begin{cases} \lnot P_k & \text{if} \;\; a_{jk} = \mathrm{T} \\ P_k & \text{if} \;\; a_{jk} = \mathrm{F} \end{cases},\]

则

\[\begin{align} H_A(x_1, x_2, \dots, x_n) = \mathrm{T} & \Leftrightarrow \hat{v}(A) = \mathrm{T} \\ & \Leftrightarrow v(P_{i,k}^{\star}) = \mathrm{T} & \text{for some } i, \text{ for all } k \\ & \Leftrightarrow v(P_{k}) = x_i = a_{ik} & \text{for some } i, \text{ for all } k \\ & \Leftrightarrow (x_1, x_2, \dots, x_n) \in T_f \\ & \Leftrightarrow f(x_1, x_2, \dots, x_n) = \mathrm{T}. \end{align}\]

类似地,

\[\begin{align} H'_A(x_1, x_2, \dots, x_n) = \mathrm{F} & \Leftrightarrow \hat{v}(A') = \mathrm{F} \\ & \Leftrightarrow v(Q_{j,k}^{\star}) = \mathrm{F} & \text{for some } j, \text{ for all } k \\ & \Leftrightarrow v(P_{k}) = x_j = b_{jk} & \text{for some } j, \text{ for all } k \\ & \Leftrightarrow (x_1, x_2, \dots, x_n) \in F_f \\ & \Leftrightarrow f(x_1, x_2, \dots, x_n) = \mathrm{F}. \end{align}\]



命题 1.15

\[\begin{align} A\simeq B & \Leftrightarrow v \vDash A \text{ iff } v \vDash B & \text{ for all } v\\ & \Leftrightarrow \hat{v}(A) = \hat{v}(B) & \text{ for all } v \\ & \Leftrightarrow H_A(v(Q_1),v(Q_2),\dots, v(Q_n)) = H_B(v(Q_1),v(Q_2),\dots, v(Q_n)) & \text{ for all } v \\ & \Leftrightarrow H_A(x_1,x_2,\dots,x_n) = H_B(x_1,x_2,\dots,x_n) & \text{ for all } \vec{x}\\ & \Leftrightarrow H_A = H_B. \end{align}\]



命题 1.16

\[H_A \xlongequal[]{1.10} f \xlongequal[]{1.12} H_B \xrightarrow{1.15} A \simeq B\]



命题 1.26

引理 3.24

\[(A[\frac{t}{x}])_{M[\sigma]} = A_{M[\sigma[x:=t_{M[\sigma]}]]} \triangleq A_{M[\rho]}\]

$ A = (u \doteq v) $

\[\begin{align} ((u \doteq v)[\frac{t}{x}])_{M[\sigma]} & = (u[\frac{t}{x}] \doteq v[\frac{t}{x}])_{M[\sigma]}\\ & = [(u[\frac{t}{x}])_{M[\sigma]} = (v[\frac{t}{x}])_{M[\sigma]}]\\ & = [u_{M[\rho]} = v_{M[\rho]}] = (u \doteq v)_{M[\rho]}. \end{align}\]

$ A = P(t_1,\dots,t_n) $

\[\begin{align} (P(\vec{t})[\frac{t}{x}])_{M[\sigma]} & = (P(t_1[\frac{t}{x}],\dots,t_n[\frac{t}{x}]))_{M[\sigma]}\\ & = [((t_1[\frac{t}{x}])_{M[\sigma]},\dots,(t_n[\frac{t}{x}])_{M[\sigma]}) \in P_M]\\ & = [((t_1)_{M[\rho]},\dots,(t_n)_{M[\rho]}) \in P_M] = (P(\vec{t}))_{M[\rho]}. \end{align}\]

$ A = \lnot B$

\[\begin{align} ((\lnot B)[\frac{t}{x}])_{M[\sigma]} & = (\lnot(B[\frac{t}{x}]))_{M[\sigma]}\\ & = \mathbf{B}_{\lnot} ((B[\frac{t}{x}])_{M[\sigma]})\\ & = \mathbf{B}_{\lnot} (B_{M[\rho]}) = (\lnot B)_{M[\rho]}. \end{align}\]

$ A = B \star C$

\[\begin{align} ((B\star C)[\frac{t}{x}])_{M[\sigma]} & = ((B[\frac{t}{x}]) \star (C[\frac{t}{x}]))_{M[\sigma]}\\ & = \mathbf{B}_{\star} ((B[\frac{t}{x}])_{M[\sigma]}, (C[\frac{t}{x}])_{M[\sigma]})\\ & = \mathbf{B}_{\star} (B_{M[\rho]}, C_{M[\rho]}) = (B \star C)_{M[\rho]}. \end{align}\]

$A = Q y. B$, $\quad$ $y\equiv x$

\[((Q y. B)[\frac{t}{x}])_{M[\sigma]} = (Q y. B)_{M[\sigma]} = (Q y. B)_{M[\rho]}.\]

$A = \forall y. B$, $\quad$ $y\not \equiv x$, $\quad$ $y\notin FV(t)$

\[\begin{align} ((\forall y. B)[\frac{t}{x}])_{M[\sigma]} & = (\forall y. B[\frac{t}{x}])_{M[\sigma]} \\ & = [(B[\frac{t}{x}])_{M[\sigma[y:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\sigma[y:=a][x:=t_{M[\sigma[y:=a]]}]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\sigma[y:=a][x:=t_{M[\sigma]}]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\sigma[x:=t_{M[\sigma]}][y:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\rho[y:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = (\forall y. B)_{M[\rho]}. \end{align}\]

$A = \forall y. B$, $\quad$ $y\not \equiv x$, $\quad$ $y\in FV(t)$

\[\begin{align} ((\forall y. B)[\frac{t}{x}])_{M[\sigma]} & = (\forall z. B[\frac{z}{y}][\frac{t}{x}])_{M[\sigma]} \\ & = [(B[\frac{z}{y}][\frac{t}{x}])_{M[\sigma[z:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [(B[\frac{z}{y}])_{M[\sigma[z:=a][x:=t_{M[\sigma[z:=a]]}]]} = \mathrm{T} & \text{ for all } a] \\ & = [(B[\frac{z}{y}])_{M[\sigma[z:=a][x:=t_{M[\sigma]}]]} = \mathrm{T} & \text{ for all } a] \\ & = [(B[\frac{z}{y}])_{M[\sigma[x:=t_{M[\sigma]}][z:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [(B[\frac{z}{y}])_{M[\rho[z:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\rho[z:=a]][y:=z_{M[\sigma[z:=a]]}]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\rho[z:=a][y:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = [B_{M[\rho[y:=a]]} = \mathrm{T} & \text{ for all } a] \\ & = (\forall y. B)_{M[\rho]}. \end{align}\]

对最后一种情形的讨论解决了习题 24.



定理 3.28

设 $t \doteq s\in \varPsi$ 与 $s \doteq u\in \varPsi$ 成立. 由 3.25(1)

\[\mathrm{Atom}(A) \text{ and } A\in \Psi\Rightarrow \lnot A\notin \Psi\]

知 $\lnot (t \doteq s) \notin \varPsi$, 进而由 3.25(3)

\[A\rightarrow B\in \varPsi \Rightarrow \lnot A \in \varPsi \text{ or } B\in \varPsi\]

与 3.25(15)

\[t \doteq s \rightarrow (s \doteq u \rightarrow t \doteq u) \in \varPsi\]

知 $(s\doteq u \rightarrow t\doteq u) \in \varPsi$. 类似可知 $t\doteq u\in \varPsi$, 即证明了传递性.



引理 3.32

1.$\quad$ $c_{M[\sigma]} = c_H = [c]$.

2.$\quad$ $x_{M[\sigma]} = \sigma(x) = [x]$.

3.$\quad$ $(f(t))_{M[\sigma]} = f_M (t _{ M[\sigma] }) = f_M([t]) = [f(t)]$.

两个项相等, 从语义上说, 即它们能通过一系列公式联系起来, 得到语法上的相等. 因此考虑关于 $\doteq$ 的等价类.



引理 3.33

$A = p(t)$

\[\begin{align} A\in \varPsi & \Rightarrow p(t)\in \varPsi \Rightarrow p_H([t]) = \mathrm{T} \Rightarrow p_H(t_H[\sigma]) = \mathrm{T} \Rightarrow (p(t))_{H[\sigma]} = \mathrm{T},\\ \lnot A\in \varPsi & \Rightarrow p(t)\notin \varPsi \Rightarrow p_H([t]) = \mathrm{F} \Rightarrow p_H(t_H[\sigma]) = \mathrm{F} \Rightarrow (p(t))_{H[\sigma]} = \mathrm{F}. \end{align}\]

$A = s \doteq t$

\[\begin{align} A\in \varPsi & \Rightarrow s\doteq t \in \varPsi \xrightarrow{3.29} [s] = [t] \Rightarrow s_{H[\sigma]} = t_{H[\sigma]} \Rightarrow (s\doteq t)_{H[\sigma]} = \mathrm{T},\\ \lnot A \in \varPsi & \Rightarrow s\doteq t \notin \varPsi \Rightarrow [s] \ne [t] \Rightarrow s_{H[\sigma]} \ne t_{H[\sigma]} \Rightarrow (s\doteq t)_{H[\sigma]} = \mathrm{F}. \end{align}\]

$A = \lnot B$

\[\begin{align} A \in \varPsi & \Rightarrow \lnot B\in \varPsi \xrightarrow{\text{I.H.}} B_{H[\sigma]} = \mathrm{F} \Rightarrow A_{H[\sigma]} = \mathbf{B}_{\lnot}(B_{H[\sigma]}) = \mathrm{T},\\ \lnot A \in \varPsi & \Rightarrow \lnot \lnot B\in \varPsi \xrightarrow{3.25(2)}B\in \varPsi \Rightarrow B_{H[\sigma]} = \mathrm{T} \Rightarrow A_{H[\sigma]} = \mathrm{F}.\\ \end{align}\]

$A = B\lor C$

\[\begin{align} A \in \varPsi & \Rightarrow B\lor C\in \varPsi \\ & \xrightarrow{3.25(7)} B\in \varPsi \text{ or } C \in \varPsi \\ & \xrightarrow{\text{I.H.}} \mathbf{B}_{\lor}(B_{H[\sigma]}, C_{H[\sigma]}) = (B \lor C)_{H[\sigma]} = \mathrm{T}, \\ \lnot A \in \varPsi & \Rightarrow \lnot(B\lor C)\in \varPsi \\ & \xrightarrow{3.25(8)} \lnot B\in \varPsi \text{ and } \lnot C \in \varPsi \\ & \Rightarrow \mathbf{B}_{\land}(\mathbf{B}_{\lnot}(B_{H[\sigma]}), \mathbf{B}_{\lnot}(C_{H[\sigma]})) = \mathrm{T} \\ & \Rightarrow \mathbf{B}_{\lor}(B_{H[\sigma]}, C_{H[\sigma]}) = \mathrm{F} \\ & \Rightarrow (B \lor C)_{H[\sigma]} = \mathrm{F}. \end{align}\]

$A = \forall x. B$

\[\begin{align} A \in \varPsi & \Rightarrow \forall x. B \in \varPsi\\ & \xrightarrow{3.25(11)} B[\frac{t}{x}] \in \varPsi & \text{for all } t \in T \\ & \xrightarrow{\text{I.H.}} (B[\frac{t}{x}])_{H[\sigma]} = \mathrm{T} & \text{for all } t \in T \\ & \xrightarrow{3.24} B_{H[\sigma[x:=t_{H[\sigma]}]]} = \mathrm{T} & \text{for all } t \in T \\ & \Rightarrow B_{H[\sigma[x:=a]]} = \mathrm{T} & \text{for all } a \in H \\ & \Rightarrow (\forall x. B)_{H[\sigma]} = A_{H[\sigma]} = \mathrm{T}, \\ \lnot A \in \varPsi & \Rightarrow \lnot (\forall x. B) \in \varPsi\\ & \xrightarrow{3.25(12)} \lnot B[\frac{t}{x}] \in \varPsi & \text{for some } t \in T \\ & \xrightarrow{\text{I.H.}} (B[\frac{t}{x}])_{H[\sigma]} = \mathrm{F} & \text{for some } t \in T \\ & \Rightarrow B_{H[\sigma[x:=t_{H[\sigma]}]]} = \mathrm{F} & \text{for some } t \in T \\ & \Rightarrow B_{H[\sigma[x:=a]]} = \mathrm{F} & \text{for some } a \in H \\ & \Rightarrow (\forall x. B)_{H[\sigma]} = A_{H[\sigma]} = \mathrm{F}. \end{align}\]
